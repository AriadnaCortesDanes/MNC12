% aquesta presició utilitzant el metode de monteCarlo ens caldria una

% ordinador normal. Podem veure que la presició que s'obté amb el mètode de

function r = romberg(f,a,b,n)
h = (b - a) ./ (2.^(0:n-1));
r(1,1) = (b - a) * (f(a) + f(b)) / 2;
for j = 2:n
subtotal = 0;
for i = 1:2^(j-2)
subtotal = subtotal + f(a + (2 * i - 1) * h(j));
end
r(j,1) = r(j-1,1) / 2 + h(j) * subtotal;
for k = 2:j
r(j,k) = (4^(k-1) * r(j,k-1) - r(j-1,k-1)) / (4^(k-1) - 1);
end
end
end
function integral = montecarlo(f, a, b, m)
% Calcula l'integral d'una funcio pel metode de montecarlo del valor
% esperat
% f -referencia a la funcio a la que fer l'integral
% m- nombre de llançaments aleatoris
% a- limit inferior
% b- limit superior
% error: de l'ordre de (1/sqrt(m)
r = rand(1,m)*(b-a) + a;
integral = (b-a)*sum(f(r))/m;
end
function [ea,er,dc,xs] = errores(x,y)
% output -- error absolut, error relatiu, decimals correctes, xifres
% significatives
% input
% x valor exacte
% y valor aproximat
ea = abs(x-y);
er = ea./abs(x);
erp = er*100;
dc = fix(-log10(2*ea));
xs = fix(-log10(2*er));
end
