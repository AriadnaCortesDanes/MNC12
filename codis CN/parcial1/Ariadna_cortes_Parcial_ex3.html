function [error, err_acumulat] = error_polinomi(coeffs, nodes)
f = @(x) x.*exp(-x.^2); % funcio real
f = f(nodes);
f_aprox = polyval(coeffs,nodes);
error = f - f_aprox;
err_acumulat = norm(error);
end
